  map.Draw(cur_x, cur_y)

PRI advance(addr, delta, w, h) : changed

  if ctrl.Right

    long[addr][PX] := w &lt;# (long[addr][PX] + delta)
  elseif ctrl.Left

  elseif ctrl.Down

    long[addr][PY] := h &lt;# (long[addr][PY] + delta)     &#39; update y of coordinate pair at addr
